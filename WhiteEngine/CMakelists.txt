cmake_minimum_required(VERSION 3.5)
project(WhiteEngine)

# Add GLAD
add_library(glad STATIC dependencies/glad/src/glad.c "src/engine.h" "src/application.h")
target_include_directories(glad PUBLIC dependencies/glad/include)

# Add GLFW
add_subdirectory(dependencies/glfw)

# Find OpenGL
find_package(OpenGL REQUIRED)

# Add executable
add_executable(WhiteEngine "src/main.cpp" "src/engine.cpp" "src/application.cpp")

# Link libraries
target_link_libraries(WhiteEngine PRIVATE glad glfw OpenGL::GL)
